1.非负整数         /^\d+$/
2.正整数           /^[0-9]*[1-9][0-9]*$/
3.非正整数         /^((-\d+)|(0+))$/
4.负整数           /^-[0-9]*[1-9][0-9]*$/
5.整数             /^-?\d+$/
6.非负浮点数       /^\d+(\.\d+)?$/
7.正浮点数         /^(([0-9]+\.[0-9]*[1-9][0-9]*)|([0-9]*[1-9][0-9]*\.[0-9]+)|([0-9]*[1-9][0-9]*))$/
8.非正浮点数       /^((-\d+(\.\d+)?)|(0+(\.0+)?))$/
9.负浮点数         /^(-(([0-9]+\.[0-9]*[1-9][0-9]*)|([0-9]*[1-9][0-9]*\.[0-9]+)|([0-9]*[1-9][0-9]*)))$/
10.浮点数          /^(-?\d+)(\.\d+)?$/
11.数字            /^\d+(\.{1}\d+)?$/
